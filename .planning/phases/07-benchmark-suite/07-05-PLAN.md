```python
def benchmark_compression_command(args):
    """Entry point for `omlx benchmark-compression` CLI subcommand."""
    import json
    from datetime import datetime, timezone

    tasks = getattr(args, "tasks", ["cosine_sim", "latency", "gsm8k", "mmlu", "litm"])
    output_path = getattr(args, "output", None)

    print(f"Running compression benchmark for {args.model}")
    print(f"Tasks: {', '.join(tasks)}")
    print(f"Seed: {args.seed}")

    runner = BenchmarkRunner(
        model_path=args.model,
        bundle_path=getattr(args, "bundle", None),
        am_ratio=getattr(args, "am_ratio", 4.0),
        seed=args.seed,
        n_samples=getattr(args, "n_samples", 200),
    )

    report = runner.run_benchmark(tasks=tasks)

    # Pretty-print summary to stdout
    print("\n--- Benchmark Results ---")
    tech = report.get("technical_metrics", {})
    qual = report.get("quality_metrics", {})
    thresh = report.get("thresholds", {})

    if "am_cosine_similarity" in tech:
        status = "PASS" if thresh.get("am_cosine_sim_pass") else "FAIL"
        print(f"  AM cosine similarity:    {tech['am_cosine_similarity']:.4f}  [{status}]")
    if "decompression_latency_ms_per_layer" in tech:
        status = "PASS" if thresh.get("latency_pass") else "FAIL"
        print(f"  Decompression latency:   {tech['decompression_latency_ms_per_layer']:.2f} ms/layer  [{status}]")
    if "gsm8k_vanilla" in qual:
        delta = abs(qual["gsm8k_vanilla"] - qual["gsm8k_compressed"])
        status = "PASS" if thresh.get("gsm8k_delta_pass") else "FAIL"
        print(f"  GSM8K: vanilla={qual['gsm8k_vanilla']:.1f}%  compressed={qual['gsm8k_compressed']:.1f}%  delta={delta:.1f}pp  [{status}]")
    if "mmlu_vanilla" in qual:
        delta = abs(qual["mmlu_vanilla"] - qual["mmlu_compressed"])
        status = "PASS" if thresh.get("mmlu_delta_pass") else "FAIL"
        print(f"  MMLU:  vanilla={qual['mmlu_vanilla']:.1f}%  compressed={qual['mmlu_compressed']:.1f}%  delta={delta:.1f}pp  [{status}]")
    if "litm_vanilla" in qual:
        delta = abs(qual["litm_vanilla"] - qual["litm_compressed"])
        status = "PASS" if thresh.get("litm_delta_pass") else "FAIL"
        print(f"  LITM:  vanilla={qual['litm_vanilla']:.3f}  compressed={qual['litm_compressed']:.3f}  delta={delta:.3f}  [{status}]")

    swa = report.get("swa_layers_skipped", [])
    if swa:
        print(f"  SWA layers skipped: {swa}")

    overall = "PASS" if report.get("overall_pass") else "FAIL"
    print(f"\nOverall: {overall}")

    # Write JSON report
    if output_path:
        with open(output_path, "w") as f:
            json.dump(report, f, indent=2)
        print(f"\nReport written to: {output_path}")
    else:
        print("\nJSON report:")
        print(json.dumps(report, indent=2))
```

```python
class TestReproducibility:
    def test_same_seed_produces_same_report(self):
        # Fast test: tasks=[] returns bare dict with deterministic fields
        r1 = BenchmarkRunner("fake/model", seed=42).run_benchmark(tasks=[])
        r2 = BenchmarkRunner("fake/model", seed=42).run_benchmark(tasks=[])
        assert r1["seed"] == r2["seed"] == 42
        assert r1["config"] == r2["config"]
        assert r1["schema_version"] == r2["schema_version"]

    @pytest.mark.slow
    def test_reproducibility_with_real_model(self):
        # Two runs with same seed on same model should produce identical technical metrics
        runner1 = BenchmarkRunner(QWEN_MODEL, seed=42)
        runner2 = BenchmarkRunner(QWEN_MODEL, seed=42)
        r1 = runner1.run_benchmark(tasks=["cosine_sim"])
        r2 = runner2.run_benchmark(tasks=["cosine_sim"])
        # Seed-controlled MLX ops should produce identical values
        assert r1["seed"] == r2["seed"]
        sim1 = r1["technical_metrics"]["am_cosine_similarity"]
        sim2 = r2["technical_metrics"]["am_cosine_similarity"]
        assert abs(sim1 - sim2) < 1e-4, f"Cosine sim not reproducible: {sim1:.6f} vs {sim2:.6f}"
```
