Function: _split_tokens(tensor, n_sink_tokens, sliding_window)
  Input: tensor [1, n_kv_heads, seq_len, head_dim]
  Compute: seq_len = tensor.shape[2]
           window_start = max(n_sink_tokens, seq_len - sliding_window)
  Return: (sinks, body, window) where
    sinks  = tensor[:, :, :n_sink_tokens, :]
    body   = tensor[:, :, n_sink_tokens:window_start, :]
    window = tensor[:, :, window_start:, :]
  Note: body may be zero-length if seq_len <= n_sink_tokens + sliding_window

Function: _calibrate_onthefly(tensor, n_components)
  Input: tensor [1, n_kv_heads, seq_len, head_dim]
  Purpose: on-the-fly PCA when no bundle available (testing path, lower quality)
  Steps:
    1. Reshape to [n_kv_heads * seq_len, head_dim], cast to float32
    2. Compute mean = mx.mean(data, axis=0)
    3. centered = data - mean
    4. Call _mx_materialize(centered) before SVD
    5. U, S, Vt = svd_f32(centered)  -- NEVER mx.linalg.svd
    6. _mx_materialize(U, S, Vt)
    7. basis = Vt[:n_components].T  -- shape [head_dim, n_components]
    8. Return (basis, mean, S[:n_components])  -- all mx.array

Function: _dp_allocate_bits(singular_values, bits_per_token_budget, n_tokens, n_components, min_bits=1, max_bits=8)
  Input: singular_values -- numpy float array [n_components], largest first
         bits_per_token_budget -- float, target bits per token
         n_tokens -- int, number of tokens in the body
         n_components -- int
  Steps:
    1. total_budget = int(bits_per_token_budget * n_tokens * n_components)
    2. alloc = np.full(n_components, min_bits, dtype=np.uint8)
    3. remaining = total_budget - min_bits * n_components
    4. priority = np.argsort(singular_values)[::-1]  -- highest SV first
    5. For each comp_idx in priority:
         extra = min(max_bits - alloc[comp_idx], remaining)
         alloc[comp_idx] += extra
         remaining -= extra
         if remaining <= 0: break
    6. Return alloc (dtype uint8, all values in [min_bits, max_bits])

Function: _lloyd_codebook(col, n_bits)
  Input: col -- numpy float array of PCA coefficients for one component
         n_bits -- int (1..8)
  Steps:
    1. n_levels = 2 ** n_bits
    2. percentiles = np.linspace(0, 100, n_levels + 1)
    3. boundaries = np.percentile(col, percentiles)
    4. centroids = (boundaries[:-1] + boundaries[1:]) / 2
    5. For 3 iterations:
         dists = np.abs(col[:, None] - centroids[None, :])
         indices = np.argmin(dists, axis=1)
         For k in range(n_levels):
           mask = indices == k
           if np.any(mask): centroids[k] = col[mask].mean()
    6. Return centroids (float64, shape [n_levels])

Function: _compress_zstd(data, level=3)
  Return: zstd.ZstdCompressor(level=level).compress(data)

Function: _decompress_zstd(compressed)
  Return: zstd.ZstdDecompressor().decompress(compressed)
