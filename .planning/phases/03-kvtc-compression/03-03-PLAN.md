<!-- Blob format (from 03-RESEARCH.md Pattern 4, Claude's Discretion on exact layout) -->

Global header (struct format "!4sIIII"):
  [4 bytes] magic b'KVTC'
  [4 bytes] version uint32 = 1
  [4 bytes] n_layers uint32
  [4 bytes] n_sink_tokens uint32
  [4 bytes] sliding_window uint32

Per-layer section (one per layer, sequential):
  [4 bytes] layer_idx uint32
  [4 bytes] n_kv_heads uint32
  [4 bytes] seq_len (original, before exemption split) uint32
  [4 bytes] head_dim uint32
  [4 bytes] n_components uint32
  [raw bytes] sink tokens float16 numpy tobytes: [1, n_kv_heads, n_sink_tokens, head_dim]
  [raw bytes] window tokens float16 numpy tobytes: [1, n_kv_heads, window_len, head_dim]
  [4 bytes] k_compressed_len uint32
  [k_compressed_len bytes] zstd-compressed K byte stream (indices + codebooks)
  [4 bytes] v_compressed_len uint32
  [v_compressed_len bytes] zstd-compressed V byte stream
  [n_components bytes] k_bit_alloc uint8 per component
  [n_components bytes] v_bit_alloc uint8 per component
  [head_dim * n_components * 4 bytes] K basis float32 (stored as numpy tobytes)
  [head_dim * 4 bytes] K mean float32
  [head_dim * n_components * 4 bytes] V basis float32
  [head_dim * 4 bytes] V mean float32

Empty body case (seq_len <= n_sink_tokens + sliding_window):
  n_components = 0
  k_compressed_len = 0, v_compressed_len = 0
  k_bit_alloc and v_bit_alloc are zero-length
  basis and mean are zero-length
  Only sinks and window stored verbatim

compress() algorithm:

1. Determine if bundle or on-the-fly path:
   - If self._bundle is not None: bundle = self._bundle
   - If self._bundle is None: use _calibrate_onthefly per layer (testing path)
   Default n_components: if no bundle, use min(64, head_dim // 2)

2. Write global header:
   - HEADER_FMT = '!4sIIII' (big-endian: magic 4s, version I, n_layers I, n_sink_tokens I, sliding_window I)
   - header_bytes = struct.pack(HEADER_FMT, b'KVTC', 1, len(kv_cache), self.n_sink_tokens, self.sliding_window)
   - layer_blobs = []

3. For each (keys, values) in kv_cache (enumerate for layer_idx):
   a. Split: sinks_k, body_k, window_k = _split_tokens(keys, self.n_sink_tokens, self.sliding_window)
             sinks_v, body_v, window_v = _split_tokens(values, self.n_sink_tokens, self.sliding_window)
   b. Get shape: n_kv_heads = keys.shape[1], seq_len = keys.shape[2], head_dim = keys.shape[3]
   c. Check empty body: body_len = body_k.shape[2]
   d. If body_len == 0:
      - n_components = 0
      - k_basis_bytes, k_mean_bytes = b'', b''
      - v_basis_bytes, v_mean_bytes = b'', b''
      - k_alloc_bytes, v_alloc_bytes = b'', b''
      - k_compressed, v_compressed = b'', b''
   e. Else (body_len > 0):
      - Determine bases:
        If self._bundle is not None: look up K_basis, K_mean, K_sv, V_basis, V_mean, V_sv from bundle per layer
        Else: call _calibrate_onthefly(body_k, n_components) -> (k_basis, k_mean, k_sv)
                   _calibrate_onthefly(body_v, n_components) -> (v_basis, v_mean, v_sv)
      - PCA project body:
        Materialize body_k and body_v: _mx_materialize(body_k, body_v)
        body_k_f32 = body_k.astype(mx.float32)
        body_v_f32 = body_v.astype(mx.float32)
        k_mean_f32 = k_mean.astype(mx.float32)
        v_mean_f32 = v_mean.astype(mx.float32)
        k_coeffs = (body_k_f32 - k_mean_f32) @ k_basis    # [1, n_kv_heads, body_len, n_components]
        v_coeffs = (body_v_f32 - v_mean_f32) @ v_basis
        _mx_materialize(k_coeffs, v_coeffs)
        k_coeffs_np = np.array(k_coeffs).reshape(-1, n_components)  # [n_kv_heads*body_len, n_components]
        v_coeffs_np = np.array(v_coeffs).reshape(-1, n_components)
      - DP allocation (use pre-baked from bundle, or compute on-the-fly):
        If bundle: k_bit_alloc = bundle['k_bit_alloc'][layer_idx]; v_bit_alloc = bundle['v_bit_alloc'][layer_idx]
        Else:
          k_sv_np = np.array(k_sv)
          v_sv_np = np.array(v_sv)
          k_bit_alloc = _dp_allocate_bits(k_sv_np, self.bits_per_token, n_kv_heads * body_len, n_components)
          v_bit_alloc = _dp_allocate_bits(v_sv_np, self.bits_per_token, n_kv_heads * body_len, n_components)
      - Quantize each component via Lloyd + pack to bytes:
        For keys:
          k_index_bytes_list = []
          k_codebook_bytes_list = []
          for c in range(n_components):
              col = k_coeffs_np[:, c]
              n_bits = int(k_bit_alloc[c])
              centroids = _lloyd_codebook(col, n_bits)
              dists = np.abs(col[:, None] - centroids[None, :])
              indices = np.argmin(dists, axis=1).astype(np.uint8)
              k_index_bytes_list.append(indices.tobytes())
              k_codebook_bytes_list.append(centroids.astype(np.float32).tobytes())
          k_index_stream = b''.join(k_index_bytes_list)
          k_codebook_stream = b''.join(k_codebook_bytes_list)
          k_raw = k_index_stream + k_codebook_stream
          k_compressed = _compress_zstd(k_raw)
        Repeat for values -> v_compressed
      - Serialize basis and mean (float32, for self-describing blob):
        _mx_materialize(k_basis, k_mean, v_basis, v_mean)
        k_basis_bytes = np.array(k_basis).astype(np.float32).tobytes()
        k_mean_bytes = np.array(k_mean.reshape(-1)).astype(np.float32).tobytes()
        v_basis_bytes = np.array(v_basis).astype(np.float32).tobytes()
        v_mean_bytes = np.array(v_mean.reshape(-1)).astype(np.float32).tobytes()
        k_alloc_bytes = k_bit_alloc.astype(np.uint8).tobytes()
        v_alloc_bytes = v_bit_alloc.astype(np.uint8).tobytes()
   f. Serialize sinks and window verbatim:
      _mx_materialize(sinks_k, window_k, sinks_v, window_v)
      # Interleave K and V for each exempt region (sinks K then sinks V, window K then window V)
      sinks_k_bytes = np.array(sinks_k.astype(mx.float16)).tobytes()
      sinks_v_bytes = np.array(sinks_v.astype(mx.float16)).tobytes()
      window_k_bytes = np.array(window_k.astype(mx.float16)).tobytes()
      window_v_bytes = np.array(window_v.astype(mx.float16)).tobytes()
      window_len = window_k.shape[2]
   g. Pack per-layer section:
      layer_header = struct.pack('!IIIII', layer_idx, n_kv_heads, seq_len, head_dim, n_components)
      window_len_packed = struct.pack('!I', window_len)
      k_comp_len_packed = struct.pack('!I', len(k_compressed))
      v_comp_len_packed = struct.pack('!I', len(v_compressed))
      layer_blob = (
          layer_header
          + sinks_k_bytes + sinks_v_bytes
          + window_len_packed + window_k_bytes + window_v_bytes
          + k_comp_len_packed + k_compressed
          + v_comp_len_packed + v_compressed
          + k_alloc_bytes + v_alloc_bytes
          + k_basis_bytes + k_mean_bytes
          + v_basis_bytes + v_mean_bytes
      )
      layer_blobs.append(layer_blob)

4. Return header_bytes + b''.join(layer_blobs)

decompress() algorithm:

1. Parse global header:
   HEADER_FMT = '!4sIIII'
   HEADER_SIZE = struct.calcsize(HEADER_FMT)
   magic, version, n_layers, n_sink_tokens, sliding_window = struct.unpack(HEADER_FMT, blob[:HEADER_SIZE])
   if magic != b'KVTC': raise ValueError(f"Invalid blob magic: {magic!r}")
   offset = HEADER_SIZE

2. For each layer (loop n_layers times):
   a. Parse layer header (struct '!IIIII' = 5 uint32 = 20 bytes):
      layer_idx, n_kv_heads, seq_len, head_dim, n_components = struct.unpack('!IIIII', blob[offset:offset+20])
      offset += 20

   b. Parse sinks K and V verbatim:
      sink_size = n_kv_heads * n_sink_tokens * head_dim  # elements
      sink_k_bytes = 2 * sink_size  # float16 = 2 bytes each
      sinks_k_np = np.frombuffer(blob[offset:offset+sink_k_bytes], dtype=np.float16).reshape(1, n_kv_heads, n_sink_tokens, head_dim)
      offset += sink_k_bytes
      sinks_v_np = np.frombuffer(blob[offset:offset+sink_k_bytes], dtype=np.float16).reshape(1, n_kv_heads, n_sink_tokens, head_dim)
      offset += sink_k_bytes

   c. Parse window length, then window K and V verbatim:
      window_len = struct.unpack('!I', blob[offset:offset+4])[0]
      offset += 4
      win_size_bytes = 2 * n_kv_heads * window_len * head_dim
      window_k_np = np.frombuffer(blob[offset:offset+win_size_bytes], dtype=np.float16).reshape(1, n_kv_heads, window_len, head_dim)
      offset += win_size_bytes
      window_v_np = np.frombuffer(blob[offset:offset+win_size_bytes], dtype=np.float16).reshape(1, n_kv_heads, window_len, head_dim)
      offset += win_size_bytes

   d. Parse compressed K body:
      k_comp_len = struct.unpack('!I', blob[offset:offset+4])[0]
      offset += 4
      k_compressed = blob[offset:offset+k_comp_len]
      offset += k_comp_len

   e. Parse compressed V body:
      v_comp_len = struct.unpack('!I', blob[offset:offset+4])[0]
      offset += 4
      v_compressed = blob[offset:offset+v_comp_len]
      offset += v_comp_len

   f. If n_components == 0 (empty body case -- short sequence):
      Skip alloc/basis/mean parsing (all zero-length).
      keys = mx.array(np.concatenate([sinks_k_np, window_k_np], axis=2))
      values = mx.array(np.concatenate([sinks_v_np, window_v_np], axis=2))
      layers.append((keys, values))
      continue

   g. Parse bit allocations:
      k_bit_alloc = np.frombuffer(blob[offset:offset+n_components], dtype=np.uint8).copy()
      offset += n_components
      v_bit_alloc = np.frombuffer(blob[offset:offset+n_components], dtype=np.uint8).copy()
      offset += n_components

   h. Parse K basis and mean (float32):
      k_basis_size = head_dim * n_components * 4  # float32
      k_basis_np = np.frombuffer(blob[offset:offset+k_basis_size], dtype=np.float32).reshape(head_dim, n_components).copy()
      offset += k_basis_size
      k_mean_np = np.frombuffer(blob[offset:offset+head_dim*4], dtype=np.float32).copy()
      offset += head_dim * 4

   i. Parse V basis and mean:
      v_basis_np = np.frombuffer(blob[offset:offset+k_basis_size], dtype=np.float32).reshape(head_dim, n_components).copy()
      offset += k_basis_size
      v_mean_np = np.frombuffer(blob[offset:offset+head_dim*4], dtype=np.float32).copy()
      offset += head_dim * 4

   j. Decompress K body:
      k_raw = _decompress_zstd(k_compressed)
      body_len = seq_len - n_sink_tokens - window_len
      n_tokens_body = n_kv_heads * body_len
      k_index_bytes = n_tokens_body * n_components  # 1 byte per index
      k_index_stream = k_raw[:k_index_bytes]
      k_codebook_stream = k_raw[k_index_bytes:]
      # Reconstruct K coefficients:
      k_coeffs_np = np.zeros((n_tokens_body, n_components), dtype=np.float32)
      codebook_offset = 0
      index_offset = 0
      for c in range(n_components):
          n_bits = int(k_bit_alloc[c])
          n_levels = 2 ** n_bits
          cb_size = n_levels * 4  # float32
          centroids = np.frombuffer(k_codebook_stream[codebook_offset:codebook_offset+cb_size], dtype=np.float32).copy()
          codebook_offset += cb_size
          indices = np.frombuffer(k_index_stream[index_offset:index_offset+n_tokens_body], dtype=np.uint8).copy()
          index_offset += n_tokens_body
          k_coeffs_np[:, c] = centroids[indices]
      k_coeffs_np = k_coeffs_np.reshape(1, n_kv_heads, body_len, n_components)

   k. Reconstruct K body: body_k_np = k_coeffs_np @ k_basis_np.T + k_mean_np  [1, n_kv_heads, body_len, head_dim]

   l. Repeat steps j-k for V -> body_v_np

   m. Concatenate: full_k_np = np.concatenate([sinks_k_np, body_k_np.astype(np.float16), window_k_np], axis=2)
      full_v_np = np.concatenate([sinks_v_np, body_v_np.astype(np.float16), window_v_np], axis=2)

   n. Convert to mx.array: keys = mx.array(full_k_np); values = mx.array(full_v_np)
   o. layers.append((keys, values))

3. Return layers
