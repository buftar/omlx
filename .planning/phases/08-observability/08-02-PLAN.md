```python
def record_compression_ratio(self, ratio: float) -> None:
    """Record a compression ratio (AM compaction ratio). Thread-safe."""
    with self._lock:
        if not hasattr(self, '_compression_ratios'):
            self._compression_ratios: List[float] = []
            self._alltime_compression_ratios: List[float] = []
        self._compression_ratios.append(ratio)
        self._alltime_compression_ratios.append(ratio)

def record_decompression_latency(self, latency_ms: float) -> None:
    """Record a decompression latency in milliseconds. Thread-safe."""
    with self._lock:
        if not hasattr(self, '_decompression_latencies'):
            self._decompression_latencies: List[float] = []
            self._alltime_decompression_latencies: List[float] = []
        self._decompression_latencies.append(latency_ms)
        self._alltime_decompression_latencies.append(latency_ms)
```

```python
def load_block(self, block_hash: bytes) -> Optional[Any]:
    import time

    # ... hot cache check ...

    entry = self._disk_cache.get(block_hash)
    if entry is None:
        return None

    start_time = time.perf_counter()

    raw_bytes = entry["data"]
    blob = PipelineBlob(
        compressed=raw_bytes,
        logical_seq_len=int(entry["file_metadata"]["logical_seq_len"]),
        compaction_ratio=float(entry["file_metadata"].get("compaction_ratio", 1.0)),  # Use actual ratio
    )

    result = self._pipeline.decompress(blob)

    end_time = time.perf_counter()
    decompression_latency_ms = (end_time - start_time) * 1000

    # Record decompression metric
    from omlx.server_metrics import get_server_metrics
    get_server_metrics().record_decompression_latency(decompression_latency_ms)

    # Update stats
    self._stats["loads"] += 1
    self._stats["decompression_success_count"] = self._stats.get("decompression_success_count", 0) + 1

    return result
```
