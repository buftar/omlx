### Task 3: JavaScript Integration

**File:** `omlx/admin/static/js/dashboard.js`

Add compression endpoint integration:

```javascript
// Compression configuration endpoint
async function updateCompressionConfig(enabled, amRatio) {
  const response = await fetch('/admin/api/compression/config', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ enabled, am_ratio: parseFloat(amRatio) })
  });
  return await response.json();
}

async function getCompressionStatus() {
  const response = await fetch('/admin/api/compression/status');
  return await response.json();
}

// Update UI from status data
function updateCompressionUI(status) {
  if (!status) return;

  const enabled = status.enabled || false;
  document.getElementById('compressionEnabled').checked = enabled;

  const ratio = status.compression_ratio || '--';
  document.getElementById('compressionRatio').textContent = ratio;

  const latency = status.decompression_latency_ms ?
    `${status.decompression_latency_ms.toFixed(2)} ms/layer` : '-- ms/layer';
  document.getElementById('decompressionLatency').textContent = latency;

  const successCount = status.success_count || 0;
  document.getElementById('compressionSuccessCount').textContent = successCount;

  const failureCount = status.failure_count || 0;
  document.getElementById('compressionFailureCount').textContent = failureCount;

  const bytesSaved = formatBytes(status.bytes_saved || 0);
  document.getElementById('bytesSaved').textContent = bytesSaved;
}

// Format bytes to human readable
function formatBytes(bytes) {
  if (bytes === 0) return '0 bytes';
  const k = 1024;
  const units = ['bytes', 'KB', 'MB', 'GB'];
  const i = Math.floor(Math.log(bytes) / Math.log(k));
  return parseFloat((bytes / Math.pow(k, i)).toFixed(2)) + ' ' + units[i];
}

// Auto-refresh status every 5 seconds
setInterval(async () => {
  const status = await getCompressionStatus();
  updateCompressionUI(status);
}, 5000);

// Save button handler
document.getElementById('saveCompressionConfig')?.addEventListener('click', async () => {
  const enabled = document.getElementById('compressionEnabled').checked;
  const amRatio = document.getElementById('amRatio').value;

  try {
    await updateCompressionConfig(enabled, amRatio);
    alert('Compression configuration saved!');
  } catch (error) {
    console.error('Failed to save compression config:', error);
    alert('Failed to save configuration');
  }
});
```
